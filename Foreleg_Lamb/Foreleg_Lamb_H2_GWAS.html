Filter the data

Firstly, as we are only looking at lambs we want to make sure we only keep records of sheep who were caught the same year they were born.

capturedatalamb <- merge(sheep,capturedata,by.x="ID")
capturedatalamb$ID <- as.factor(capturedatalamb$ID)
capturedatalamb <- capturedatalamb[which(capturedatalamb$CapYear == capturedatalamb$BirthYear),]

Now, only include records of capture events that measured the foreleg.

capturedatalamb <- capturedatalamb[!is.na(capturedatalamb$Foreleg),]

We are only interested in the capture events that happened in August (to ensure the lambs are all the same age)

capturedatalamb<- capturedatalamb[which(capturedatalamb$CapMonth == 8),]

Calculate the age of the sheep in days of how old they are for their foreleg measure

capturedatalamb <- capturedatalamb[!is.na(capturedatalamb$BirthMonth),]
capturedatalamb <- capturedatalamb[!is.na(capturedatalamb$BirthDay),]
capturedatalamb$BirthDate <- paste0(capturedatalamb$BirthYear,"/",capturedatalamb$BirthMonth,"/",capturedatalamb$BirthDay)

capturedatalamb$Date <- as.numeric(format(as.Date(capturedatalamb$BirthDate),"%j"))
capturedatalamb$CapDate <- paste0(capturedatalamb$CapYear,"/",capturedatalamb$CapMonth,"/",capturedatalamb$CapDay)
capturedatalamb$AgeDate <- as.numeric(format(as.Date(capturedatalamb$CapDate),"%j"))

capturedatalamb$AugCapAge <- capturedatalamb$AgeDate - capturedatalamb$Date
capturedatalamb$Sex <- as.factor(capturedatalamb$Sex)

Find the population density the year of the measurement

Villdens <- sqlFetch(con,"HirtaVillPop")
capturedatalamb <- merge(capturedatalamb,Villdens[,c("VillTotal","Year")],by.x="BirthYear",by.y="Year",all.x=T)

And remove those that do not have their mother recorded.

capturedatalamb <- capturedatalamb[!is.na(capturedatalamb$MumID),]


To ensure the same individuals are used for all three heritability estimates, remove any sheep that are not included in the genotype data

SNPs_50K <- read.table("~/Documents/Sheep/50KSNP/Plates_1to87_QC3.fam")

#!/bin/sh
#$-cwd
#$-l h_rt=1:00:00
#$-pe sharedmem 1
#$-l h_vmem=100G
#$-M s1944058@sms.ed.ac.uk
#$-m baes
. /etc/profile.d/modules.sh

module load igmm/apps/dissect/1.15.2c
module load igmm/apps/plink/1.90b4

plink file ../InputData/Plates_1to87_QC3 --ped LambForeleg50K.ped --sheep --not-chr 0,X --make-bed --out 50K/LambForeleg50K

mpirun -np 8 dissect.mpich --reml --bfile 50K/LambForeleg50K --pheno LambForelegPhenotypes.txt --pheno-col 2 --blue --covar LambForelegAssoc.txt --qcovar LambForelegQAssoc.txt --random-effects LambForelegRandom.txt --random-effects-cols 1 2 --out 50K/LambForeLeg_50K_H2

if [[ -f 50K/LambForeLeg_50K_H2.badsnps ]]
then
plink --bfile ../InputData/Plates_1to87_QC3 --sheep --exclude 50K/LambForeLeg_50K_H2.badsnps --make-bed --out 50K/LambForeLeg_50K_H2

mpirun -np 8 dissect.mpich --reml --bfile 50K/LambForeLeg_50K_H2 --pheno LambForelegPhenotypes.txt --pheno-col 2 --blue --covar LambForelegAssoc.txt --qcovar LambForelegQAssoc.txt --random-effects LambForelegRandom.txt --random-effects-cols 1 2 --out 50K/LambForeLeg_50K_H2
fi

mpirun -np 8 dissect.mpich --reml --bfile 50K/LambForeleg50K --pheno LambForelegPhenotypes.txt --pheno-col 2 --blue --random-effects LambForelegRandom.txt --random-effects-cols 1 2 --out 50K/LambForeLeg_50K_TotalPhenotypeVariance

#!/bin/sh
#$-cwd
#$-l h_rt=2:00:00
#$-pe sharedmem 1
#$-l h_vmem=200G
#$-t 1-26
#$-tc 5
#$-M s1944058@sms.ed.ac.uk
#$-m baes
. /etc/profile.d/modules.sh

module load igmm/apps/dissect/1.15.2c
module load igmm/apps/plink/1.90b4

i=$_TASK_ID

plink --bfile ../InputData/Plates_1to87_QC3 --ped LambForeleg50K.ped --sheep --not-chr ${i},X --keep LambForelegPhenotypes.txt --make-bed --out 50K/NoChr${i}

mpirun -np 8 dissect.mpich --make-grm --bfile 50K/NoChr${i} --out 50K/NoChr${i}_GRM

echo "50K/Chr${i} 50K/NoChr${i}_GRM" > 50K/GWAS_GRM${i}.txt

plink --bfile ../InputData/Plates_1to87_QC3 --ped LambForeleg50K.ped --sheep --chr ${i} --keep LambForelegPhenotypes.txt --make-bed --out 50K/Chr${i}

if [[ -f 50K/NoChr${i}_GRM.badsnps ]]
then
plink --bfile 50K/NoChr${i} --sheep --exclude 50K/NoChr${i}_GRM.badsnps --make-bed --out 50K/NoChr${i}_2

mpirun -np 8 dissect.mpich --make-grm --bfile 50K/NoChr${i}_2 --out 50K/NoChr${i}_GRM

fi

mpirun -np 8 dissect.mpich --gwas --bfile-grm-list 50K/GWAS_GRM${i}.txt --pheno LambForelegPhenotypes.txt --pheno-col 2 --covar LambForelegAssoc.txt --qcovar LambForelegQAssoc.txt --random-effects LambForelegRandom.txt --random-effects-cols 1 2 --out 50K/LambForeLeg_50K_GWAS_Chr${i}

if [[ -f 50K/LambForeLeg_50K_GWAS_Chr${i}.badsnps ]]
then
plink --bfile 50K/Chr${i} --sheep --exclude 50K/LambForeLeg_50K_GWAS_Chr${i}.badsnps --make-bed --out 50K/Chr${i}_2

echo "50K/Chr${i}_2 50K/NoChr${i}_GRM" > 50K/GWAS_GRM${i}.txt

mpirun -np 8 dissect.mpich --gwas --bfile-grm-list 50K/GWAS_GRM${i}.txt --pheno LambForelegPhenotypes.txt --pheno-col 2 --covar LambForelegAssoc.txt --qcovar LambForelegQAssoc.txt --random-effects LambForelegRandom.txt --random-effects-cols 1 2 --out 50K/LambForeLeg_50K_GWAS_Chr${i}
fi

files = list.files(pattern="LambForeLeg_50K_GWAS_Chr",recursive=TRUE)

SNPresults = do.call(rbind, lapply(files, function(x) read.table(x,header=T)))
SNPresults$Effect <- SNPresults$STDEV*SNPresults$STDEV*SNPresults$BETA*SNPresults$BETA
bimFile <- read.table("C:/Users/s1944058/Documents/Sheep/50KSNP/Plates_1to87_QC3.bim",header=F)

GWAS50KResults <- merge(bimFile,SNPresults[,c("SNP","PV","Effect"),],by.x="V2",by.y="SNP",all.x=FALSE,all.y=TRUE)
colnames(GWAS50KResults) <- c("SNP","Chr","V3","Pos","Ref","Minor","PV","Effect")
GWAS50KResults$Chr <- as.numeric(GWAS50KResults$Chr)
par(mfrow=c(1,2))
manhattan(GWAS50KResults,chr="Chr",bp="Pos",p="PV",snp="SNP", main = "Lamb foreleg length Manhattan plot (50K)", genomewideline = -log10(0.05/20082),suggestiveline = FALSE)
qq(GWAS50KResults$PV,main="Lamb foreleg length QQ plot (50K)")

SNPs <- GWAS50KResults[which(GWAS50KResults$PV < (0.05/20082) ),c("Chr","Pos","SNP","PV","Effect")]
SNPs[order(SNPs$PV),]
